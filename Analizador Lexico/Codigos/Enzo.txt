Una_Bella_Hamburguesa
{
    El comestible Puntual;
    El precio Puntual;
    Se_escribe "Ingresa cuantas hamburguesas quieres: ";
    Escucha -el comestible;
    Si -el comestible Envidia -a 2 Entonces
    {
        -el precio Dice comestible Abraza -a 120
        Se_escribe "El precio total es:", precio
    }
    Si -el comestible Envidia -a 5 Entonces
    {
        -el precio Dice comestible Abraza -a 110
        Se_escribe "Oferta Especial! El precio total es:", precio
    }
    Sino
    {
        -el precio Dice comestible Abraza -a 100
        Se_escribe "Oferta Super Especial! El precio total es:", precio
    }
}
Fin
